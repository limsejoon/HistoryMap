기존 `// TODO` 주석을 전부 아래 코드로 교체:

```js
const WORDS = [
  { front: "사과",  back: "apple"  },
  { front: "책",   back: "book"   },
  { front: "물",   back: "water"  },
  { front: "친구", back: "friend" },
  { front: "집",   back: "house"  },
];

let cards = [];
let current = 0;
let flipped = false;
let score = { know: 0, dontKnow: 0 };

function shuffle(arr) {
  const a = [...arr];
  for (let i = a.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [a[i], a[j]] = [a[j], a[i]];
  }
  return a;
}

function renderCard() {
  const w = cards[current];
  document.getElementById('frontWord').textContent = w.front;
  document.getElementById('backLabel').textContent = w.front;
  document.getElementById('backAnswer').textContent = w.back;

  const pct = ((current + 1) / cards.length) * 100;
  document.getElementById('progressText').textContent = `${current + 1} / ${cards.length}`;
  document.getElementById('progressFill').style.width = pct + '%';

  document.getElementById('knowCount').textContent = `✓ 알았어: ${score.know}`;
  document.getElementById('dontknowCount').textContent = `✗ 몰랐어: ${score.dontKnow}`;
}

function init() {
  cards = shuffle(WORDS);
  current = 0;
  flipped = false;
  score = { know: 0, dontKnow: 0 };

  document.getElementById('card').classList.remove('flipped');
  document.getElementById('buttons').classList.remove('visible');
  document.getElementById('quizScreen').classList.remove('hidden');
  document.getElementById('doneScreen').classList.remove('visible');

  renderCard();
}

init();
```

```js
function answer(knew) {
  if (knew) {
    score.know++;
  } else {
    score.dontKnow++;
  }

  current++;

  if (current >= cards.length) {
    showDone();
    return;
  }

  // 카드 리셋 후 다음 문제
  flipped = false;
  document.getElementById('card').classList.remove('flipped');
  document.getElementById('buttons').classList.remove('visible');

  // 플립 애니메이션이 끝난 뒤 내용 교체 (0.5s transition)
  setTimeout(renderCard, 500);
}
```
